function isIsogram(str) {
   let strNew = str.replace(/[^A-Z]/gi, '').toLowerCase();
   let map = {};
   for (let i = 0; i < strNew.length; i++) {
      let symbol = strNew[i];
      map[symbol] = (map[symbol] || 0) + 1;
      if (map[symbol] > 1) 
         { 
            return false;
         }
      }
   return true;
}
